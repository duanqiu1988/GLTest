#version 300 es
layout (location = 0) in vec3 position;
layout (location = 1) in vec3 color;
layout (location = 2) in vec2 texCoord;
out vec2 TexCoord;
out vec3 outColor;

void main() {
    gl_Position = vec4(position, 1f);
    TexCoord = vec2(texCoord.x, 1.0f - texCoord.y);
    outColor = color;
    }
